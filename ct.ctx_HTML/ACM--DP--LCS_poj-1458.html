//最长公共子序列 LCS
//d[i][j] ：第一个字符串前i个 和 第二个字符串前j个 的LCS 的 最长的 长度
//d[i][j] = max(d[i][j-1], d[i-1][j]);
//if(s[i] = s2[j]) d[i][j] ++;

#include<iostream>
#include<cstdio>
#include<cctype>
#include<cstring>
#include<algorithm>
using namespace std;

const int maxn = 1e3+3;
char s1[maxn], s2[maxn], c;
int d[maxn][maxn], l1, l2;

void Read(char *s);//??

int main(){
while(true){
memset(d, 0, sizeof d);
memset(s1, 0, sizeof s1);
memset(s2, 0, sizeof s2);

Read(s1+1); if(c == EOF) break;
l1 = strlen(s1+1); 
Read(s2+1); l2 = strlen(s2+1);

for(int i=1; i<=l1; i++){
for(int j=1; j<= l2; j++){
int &ans = d[i][j];
ans = max(d[i-1][j], d[i][j-1]);
if(s1[i] == s2[j]) ans = max(ans, d[i-1][j-1]+1);
}
}
cout <<d[l1][l2] <<endl;
}
return 0;
}
void Read(char *s){
int l=0;
c = getchar();
while(!isalpha(c)){
if(c == EOF) return;
c = getchar();
}
while(isalpha(c)){
s[l++] = c, c=getchar();
}
}
